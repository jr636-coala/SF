create table AccountType(type varchar(20) NOT NULL UNIQUE, PRIMARY KEY (type));
create table User(id int AUTO_INCREMENT NOT NULL UNIQUE, username varchar(50) NOT NULL UNIQUE, password char(64) NOT NULL, creationDate date NOT NULL, type varchar(20), PRIMARY KEY (id), CONSTRAINT FK_AccountType FOREIGN KEY (type) REFERENCES AccountType(type));
create table Post(id int AUTO_INCREMENT NOT NULL UNIQUE, title text NOT NULL, content text NOT NULL, userId int, creationDate date NOT NULL, creationTime datetime NOT NULL, PRIMARY KEY (id), CONSTRAINT FK_User FOREIGN KEY (userId) REFERENCES User(id));
